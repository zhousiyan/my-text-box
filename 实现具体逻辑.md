1. 先是文本框的input事件  
~~~js
        this.editor.addEventListener('input', (event)=>{
            let tag =  event.target;
            let v = tag.value;
            let intercept = this.editor.value;
            //判断输出的字段

            this.tentative(intercept);

        
            let pos =  this.getlocation(tag);
~~~
